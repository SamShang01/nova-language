class FieldAccess:

        self.access = access
        self.default_value = default_value
        self.name = None
    
    def __set_name__(self, owner, name):
        self.name = name
        # 注册访问控制信息
        if not hasattr(owner, '_c_access_control'):
            owner._c_access_control = {}
        owner._c_access_control[name] = self.access
    
    def __get__(self, obj, objtype=None):
        if obj is None:
            return self
        
        # 直接进行访问控制检查
        self._check_access(obj, 'read')
        
        # 返回实际存储的值

        if hasattr(obj, internal_name):
            return getattr(obj, internal_name)
        elif self.default_value is not None:
            return self.default_value
        else:
            raise AttributeError(f"Field '{self.name}' not initialized")
    
    def __set__(self, obj, value):
        # 直接进行访问控制检查

        # 设置实际存储的值

    def _check_access(self, obj, operation):
        """直接检查字段访问权限"""
        if self.access == 'public':
            return  # 公开字段可以任意访问
        
        # 获取调用栈信息
        import traceback
        stack = traceback.extract_stack()
        
        # 查找真正的调用者（跳过描述符相关的帧）

            try:
                frame_info = stack[caller_frame_index]
                caller_self = None
                
                # 检查调用者是否在类的方法中

                            return  # 允许访问
                        else:
                            raise CAccessError(
                                f"Cannot {operation} protected field '{self.name}' from {caller_class.__name__}"
                            )
                
                caller_frame_index -= 1
                
            except IndexError:
                break
        
        # 如果没有找到合适的调用者，拒绝非公开访问
        if self.access != 'public':
            raise CAccessError(
                f"Cannot {operation} {self.access} field '{self.name}' from module level"
            )

def _c_check_access_impl_method(self, attr_name: str) -> None:
    """检查属性访问权限的实现方法 - 修复版本"""
    # 特殊方法豁免 - Python 内部机制需要访问这些方法
    special_methods = {'__repr__', '__str__', '__format__', '__hash__', 
                      '__bool__', '__bytes__', '__dir__', '__len__',
                      '__getitem__', '__setitem__', '__delitem__',
                      '__iter__', '__next__', '__reversed__',
                      '__contains__', '__call__', '__enter__', '__exit__'}
    
    if attr_name in special_methods:
        return  # 允许 Python 内部机制访问特殊方法
    
    access_control = getattr(self, '_c_access_control', {})
    if attr_name in access_control:
        access_level = access_control[attr_name]
        
        if access_level == 'public':
            return  # 公开属性可以访问
        
        # 获取调用栈信息
        import traceback
        stack = traceback.extract_stack()
        
        # 查找真正的调用者（跳过当前检查方法）
        for frame_info in reversed(stack[:-2]):  # 跳过当前方法和调用者
            # 检查调用者是否在类的方法中
            caller_self = None
            for var_name, var_value in frame_info.locals.items():
                if var_name == 'self' and hasattr(var_value, '__class__'):
                    caller_self = var_value
                    break
            
            if caller_self is not None:
                caller_class = caller_self.__class__
                
                if access_level == 'private':
                    # private: 只有当前类可以访问
                    if caller_class == self.__class__:
                        return  # 允许访问
                    else:
                        raise CAccessError(
                            f"Cannot access private attribute '{attr_name}' from {caller_class.__name__}"
                        )
                
                elif access_level == 'protected':
                    # protected: 当前类和子类可以访问
                    if issubclass(caller_class, self.__class__):
                        return  # 允许访问
                    else:
                        raise CAccessError(
                            f"Cannot access protected attribute '{attr_name}' from {caller_class.__name__}"
                        )
        
        # 如果没有找到合适的调用者，拒绝非公开访问
        raise CAccessError(
            f"Cannot access {access_level} attribute '{attr_name}' from module level"
        )
