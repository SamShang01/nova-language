示例 Trait系统

Trait系统示例

学习如何使用 Nova 的 Trait 系统实现多态和代码复用

概述

Trait（特质）是 Nova 实现代码复用和多态的核心机制。通过定义和实现 Trait，可以为不同类型提供统一的行为接口。

定义Trait

使用 trait 关键字定义行为接口

实现Trait

使用 impl 关键字为类型实现 Trait

Trait约束

在泛型中使用 Trait 限制类型

示例代码 

// 定义特质
trait Display {
    func display(self: Self) -> string;
}

trait Comparable {
    func compare(self: Self, other: Self) -> int;
}

trait Clone {
    func clone(self: Self) -> Self;
}

// 为基本类型实现特质
impl Display for int {
    func display(self: int) -> string {
        return str(self);
    }
}

// 定义结构体
struct Point {
    x: int;
    y: int;
}

impl Display for Point {
    func display(self: Point) -> string {
        return f"Point({self.x}, {self.y})";
    }
}

impl Clone for Point {
    func clone(self: Point) -> Point {
        return Point(x: self.x, y: self.y);
    }
}

// 泛型函数，使用特质约束
generic func print_display<T>(value: T)
    where T: Display
{
    print(value.display());
}

// 使用示例
func main() {
    let x = 42;
    print_display(x);  // 输出: 42
    
    let p = Point(x: 10, y: 20);
    print_display(p);  // 输出: Point(10, 20)
}
