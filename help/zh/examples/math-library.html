数学库示例

学习如何使用 Nova 标准数学库进行数学计算

概述

Nova 的标准数学库提供了丰富的数学函数和常量，包括三角函数、指数函数、对数函数、取整函数等。

数学常量

PI, E 等常用数学常量

sin, cos, tan, asin, acos, atan

指数与对数

pow, sqrt, exp, log, log10

取整函数

floor, ceil, round, abs

示例代码 

use std.math;

// 计算圆的面积
func circle_area(radius: float) -> float {
    return PI * radius * radius;
}

// 计算两点之间的距离
func distance(x1: float, y1: float, x2: float, y2: float) -> float {
    let dx = x2 - x1;
    let dy = y2 - y1;
    return sqrt(dx * dx + dy * dy);
}

// 计算阶乘
func factorial(n: int) -> int {
    if n <= 1 {
        return 1;
    }
    return n * factorial(n - 1);
}

// 判断是否为质数
func is_prime(n: int) -> bool {
    if n <= 1 return false;
    if n <= 3 return true;
    if n % 2 == 0 || n % 3 == 0 return false;
    let i = 5;
    while i * i <= n {
        if n % i == 0 || n % (i + 2) == 0 {
            return false;
        }
        i = i + 6;
    }
    return true;
}

// 主函数
func main() {
    print("PI = " + PI);
    print("sqrt(16) = " + sqrt(16));
    print("pow(2, 10) = " + pow(2, 10));
    print("sin(45°) = " + sin(PI / 4));
}
