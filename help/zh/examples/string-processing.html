字符串处理示例

学习如何使用 Nova 字符串处理函数进行文本操作

概述

Nova 提供了丰富的字符串处理函数，包括大小写转换、修剪、分割、连接、替换、子串提取等操作。

基本操作

to_upper, to_lower, trim, len

分割连接

split, join, substring

查找替换

contains, replace, find

示例代码 

use std.string;

// 反转字符串
func reverse_string(s: string) -> string {
    let chars = split(s, "");
    let result = "";
    for i in [chars.len() - 1, 0, -1] {
        result = result + chars[i];
    }
    return result;
}

// 检查回文字符串
func is_palindrome(s: string) -> bool {
    let trimmed = trim(to_lower(s));
    let reversed = reverse_string(trimmed);
    return trimmed == reversed;
}

// 首字母大写
func capitalize(s: string) -> string {
    if len(s) == 0 return s;
    let first = to_upper(substring(s, 0, 1));
    let rest = substring(s, 1, len(s));
    return first + rest;
}

// 主函数
func main() {
    let s = "Hello, World!";
    print(to_upper(s));     // HELLO, WORLD!
    print(to_lower(s));     // hello, world!
    
    let fruits = split("apple,banana,orange", ",");
    print(join(fruits, " - ")); // apple - banana - orange
    
    print(replace(s, "World", "Nova")); // Hello, Nova!
    
    print(is_palindrome("racecar")); // true
}
