STL库测试示例

学习如何使用 Nova 标准模板库（STL）中的容器和算法

概述

Nova 标准模板库（STL）提供了丰富的数据结构和算法，包括 Vector、List、Stack、Queue、Set、Map 等容器。

序列容器

Vector, List, Stack, Queue

关联容器

Set, Map

算法

sort, find, filter, map

示例代码 

func main() {
    print("=== Nova STL Library Test Suite ===\n");
    testVector();
    testMap();
    testAlgorithms();
}

func testVector() {
    var vec = Vector<int>();
    vec.push(1);
    vec.push(2);
    vec.push(3);
    
    print("Vector: " + vec.toArray().toString());
    print("Size: " + vec.count());
    print("Contains 2: " + vec.contains(2));
}

func testMap() {
    var map = Map<string, int>();
    map.insert("one", 1);
    map.insert("two", 2);
    
    print("Get 'two': " + map.get("two"));
    print("Keys: " + map.keys().toArray().toString());
}

func testAlgorithms() {
    var vec = Vector<int>();
    vec.push(5);
    vec.push(2);
    vec.push(8);
    
    sort(vec);
    print("After sort: " + vec.toArray().toString());
    
    var filtered = filter(vec, func(x: int): bool {
        return x > 3;
    });
    print("Filtered (>3): " + filtered.toArray().toString());
}
