运算符

在 Nova 中使用各种运算符

首页 > 教程 > 运算符

算术运算符

算术运算符用于执行基本的数学运算。

// 算术运算符 let a = 10; let b = 5; print("a + b = " + (a + b)); // 加法: 15 print("a - b = " + (a - b)); // 减法: 5 print("a * b = " + (a * b)); // 乘法: 50 print("a / b = " + (a / b)); // 除法: 2 print("a % b = " + (a % b)); // 取模: 0 print("a ** b = " + (a ** b)); // 幂运算: 100000 // 自增和自减 let x = 5; print(x++); // 后自增: 5 (先使用后增加) print(++x); // 前自增: 7 (先增加后使用) let y = 5; print(y--); // 后自减: 5 (先使用后减少) print(--y); // 前自减: 3 (先减少后使用)

比较运算符

比较运算符用于比较两个值并返回布尔值。

// 比较运算符 let a = 10; let b = 5; print("a == b: " + (a == b)); // 等于: false print("a != b: " + (a != b)); // 不等于: true print("a > b: " + (a > b)); // 大于: true print("a < b: " + (a < b)); // 小于: false print("a >= b: " + (a >= b)); // 大于等于: true print("a <= b: " + (a <= b)); // 小于等于: false // 字符串比较 let str1 = "apple"; let str2 = "banana"; print("str1 == str2: " + (str1 == str2)); // false print("str1 < str2: " + (str1 < str2)); // true (按字典序比较)

逻辑运算符

逻辑运算符用于组合布尔值。

// 逻辑运算符 let a = true; let b = false; print("a && b: " + (a && b)); // 逻辑与: false print("a || b: " + (a || b)); // 逻辑或: true print("!a: " + !a); // 逻辑非: false // 短路评估 let x = 5; let y = 0; // 逻辑与短路: 当第一个操作数为 false 时，不评估第二个操作数 if (y !== 0 && x / y > 0) { print("不会执行这里"); } // 逻辑或短路: 当第一个操作数为 true 时，不评估第二个操作数 if (x > 0 || y / x > 0) { print("会执行这里"); }

赋值运算符

赋值运算符用于将值赋给变量。

// 赋值运算符 let a = 10; // 基本赋值 // 复合赋值 let b = 5; b += 3; // 等价于 b = b + 3 print("b += 3: " + b); // 8 b -= 2; // 等价于 b = b - 2 print("b -= 2: " + b); // 6 b *= 2; // 等价于 b = b * 2 print("b *= 2: " + b); // 12 b /= 3; // 等价于 b = b / 3 print("b /= 3: " + b); // 4 b %= 3; // 等价于 b = b % 3 print("b %= 3: " + b); // 1 b **= 2; // 等价于 b = b ** 2 print("b **= 2: " + b); // 1

一元运算符

一元运算符只需要一个操作数。

// 一元运算符 let a = 5; print("+a: " + +a); // 一元加号: 5 print("-a: " + -a); // 一元减号: -5 let b = true; print("!b: " + !b); // 逻辑非: false // 类型运算符 print(typeof 42); // "number" print(typeof "hello"); // "string" print(typeof true); // "boolean" print(typeof null); // "object" print(typeof undefined); // "undefined" print(typeof []); // "array" print(typeof {}); // "object" print(typeof function() {}); // "function"

位运算符

位运算符用于对二进制位进行操作。

// 位运算符 let a = 6; // 二进制: 110 let b = 3; // 二进制: 011 print("a & b: " + (a & b)); // 按位与: 2 (010) print("a | b: " + (a | b)); // 按位或: 7 (111) print("a ^ b: " + (a ^ b)); // 按位异或: 5 (101) print("~a: " + ~a); // 按位非: -7 print("a << 1: " + (a << 1)); // 左移: 12 (1100) print("a >> 1: " + (a >> 1)); // 右移: 3 (11) print("a >>> 1: " + (a >>> 1)); // 无符号右移: 3 (11)

字符串运算符

字符串运算符用于操作字符串。

// 字符串运算符 let str1 = "Hello"; let str2 = "World"; // 字符串拼接 let result = str1 + " " + str2; print(result); // "Hello World" // 字符串重复 let repeated = "Hi" * 3; print(repeated); // "HiHiHi" // 字符串比较 print("apple" < "banana"); // true print("apple" === "apple"); // true

数组和对象运算符

数组和对象运算符用于访问和操作数组和对象。

// 数组运算符 let arr = [1, 2, 3, 4, 5]; // 访问数组元素 print(arr[0]); // 1 print(arr[2]); // 3 // 修改数组元素 arr[1] = 10; print(arr); // [1, 10, 3, 4, 5] // 对象运算符 let person = { name: "John", age: 30 }; // 点表示法访问 print(person.name); // "John" // 方括号表示法访问 print(person["age"]); // 30 // 修改对象属性 person.age = 31; person["email"] = "john@example.com"; print(person); // { name: "John", age: 31, email: "john@example.com" } // 检查属性是否存在 print("name" in person); // true print("phone" in person); // false

三元运算符

三元运算符是一种简洁的条件表达式。

// 三元运算符 let age = 18; let message = age >= 18 ? "成年人" : "未成年人"; print(message); // "成年人" // 嵌套三元运算符 let score = 85; let grade = score >= 90 ? "A" : score >= 80 ? "B" : score >= 70 ? "C" : score >= 60 ? "D" : "F"; print(grade); // "B"

运算符优先级

运算符优先级决定了表达式中运算符的执行顺序。

  • 1. 括号: ()
  • 2. 一元运算符: +, -, !, ~
  • 3. 幂运算: **
  • 4. 乘除取模: *, /, %
  • 5. 加减: +, -
  • 6. 位运算符: <<, >>, >>>, &, ^, |
  • 7. 比较运算符: <, <=, >, >=, ==, !=, ===, !==
  • 8. 逻辑与: &&
  • 9. 逻辑或: ||
  • 10. 三元运算符: ? :
  • 11. 赋值运算符: =, +=, -=, *=, /=, %=, **=
// 运算符优先级示例 let result = 10 + 5 * 2; // 先乘后加 print(result); // 20 let result2 = (10 + 5) * 2; // 先括号内 print(result2); // 30 let result3 = 10 > 5 && 5 < 3; // 先比较后逻辑与 print(result3); // false

练习

尝试完成以下练习，以巩固你对运算符的理解：

  1. 使用算术运算符计算圆的面积和周长
  2. 使用比较运算符和逻辑运算符检查一个数是否在某个范围内
  3. 使用位运算符实现两个数的交换
  4. 使用三元运算符根据分数返回等级
// 练习答案示例 // 1. 计算圆的面积和周长 let radius = 5; let pi = 3.14159; let area = pi * radius ** 2; let perimeter = 2 * pi * radius; print("圆的面积: " + area); print("圆的周长: " + perimeter); // 2. 检查数是否在范围内 def isInRange(num: number, min: number, max: number) -> boolean { return num >= min && num <= max; } print(isInRange(5, 1, 10)); // true print(isInRange(15, 1, 10)); // false // 3. 使用位运算符交换两个数 def swap(a: int, b: int) -> (int, int) { a = a ^ b; b = a ^ b; a = a ^ b; return (a, b); } let x = 10; let y = 20; print("交换前: x = " + x + ", y = " + y); let (newX, newY) = swap(x, y); print("交换后: x = " + newX + ", y = " + newY); // 4. 使用三元运算符返回等级 def getGrade(score: number) -> string { return score >= 90 ? "A" : score >= 80 ? "B" : score >= 70 ? "C" : score >= 60 ? "D" : "F"; } print(getGrade(95)); // A print(getGrade(85)); // B print(getGrade(75)); // C print(getGrade(65)); // D print(getGrade(55)); // F
