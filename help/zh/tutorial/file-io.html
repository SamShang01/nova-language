文件 I/O

在 Nova 中处理文件操作

首页 > 教程 > 文件 I/O

文件模块导入

在 Nova 中，你需要导入 fs 模块来进行文件操作。

// 导入文件系统模块 import fs from "fs";

读取文件

你可以使用 fs.readFileSync 来同步读取文件内容，或使用 fs.readFile 来异步读取文件内容。

// 同步读取文件 import fs from "fs"; try { let content = fs.readFileSync("example.txt", "utf8"); print("文件内容:"); print(content); } catch (e) { print("读取文件时发生错误: " + e.message); } // 异步读取文件 fs.readFile("example.txt", "utf8", (err, content) => { if (err) { print("读取文件时发生错误: " + err.message); return; } print("文件内容:"); print(content); });

写入文件

你可以使用 fs.writeFileSync 来同步写入文件，或使用 fs.writeFile 来异步写入文件。

// 同步写入文件 import fs from "fs"; try { fs.writeFileSync("output.txt", "Hello, Nova!"); print("文件写入成功"); } catch (e) { print("写入文件时发生错误: " + e.message); } // 异步写入文件 fs.writeFile("output.txt", "Hello, Nova!", (err) => { if (err) { print("写入文件时发生错误: " + err.message); return; } print("文件写入成功"); });

追加文件

你可以使用 fs.appendFileSync 来同步追加文件内容，或使用 fs.appendFile 来异步追加文件内容。

// 同步追加文件 import fs from "fs"; try { fs.appendFileSync("output.txt", "\n追加的内容"); print("文件追加成功"); } catch (e) { print("追加文件时发生错误: " + e.message); } // 异步追加文件 fs.appendFile("output.txt", "\n追加的内容", (err) => { if (err) { print("追加文件时发生错误: " + err.message); return; } print("文件追加成功"); });

删除文件

你可以使用 fs.unlinkSync 来同步删除文件，或使用 fs.unlink 来异步删除文件。

// 同步删除文件 import fs from "fs"; try { fs.unlinkSync("file-to-delete.txt"); print("文件删除成功"); } catch (e) { print("删除文件时发生错误: " + e.message); } // 异步删除文件 fs.unlink("file-to-delete.txt", (err) => { if (err) { print("删除文件时发生错误: " + err.message); return; } print("文件删除成功"); });

检查文件是否存在

你可以使用 fs.existsSync 来检查文件是否存在。

// 检查文件是否存在 import fs from "fs"; if (fs.existsSync("example.txt")) { print("文件存在"); } else { print("文件不存在"); }

文件信息

你可以使用 fs.statSync 来获取文件的详细信息。

// 获取文件信息 import fs from "fs"; try { let stats = fs.statSync("example.txt"); print("文件大小: " + stats.size + " 字节"); print("是否为文件: " + stats.isFile()); print("是否为目录: " + stats.isDirectory()); print("创建时间: " + stats.birthtime); print("修改时间: " + stats.mtime); } catch (e) { print("获取文件信息时发生错误: " + e.message); }

目录操作

你可以使用 fs.mkdirSync 创建目录，使用 fs.readdirSync 读取目录内容，使用 fs.rmdirSync 删除目录。

// 目录操作 import fs from "fs"; // 创建目录 try { fs.mkdirSync("new-directory"); print("目录创建成功"); } catch (e) { print("创建目录时发生错误: " + e.message); } // 读取目录内容 try { let files = fs.readdirSync("."); print("当前目录文件:"); for (let file in files) { print(file); } } catch (e) { print("读取目录时发生错误: " + e.message); } // 删除目录 try { fs.rmdirSync("new-directory"); print("目录删除成功"); } catch (e) { print("删除目录时发生错误: " + e.message); }

二进制文件操作

你可以读取和写入二进制文件，例如图片、音频等。

// 二进制文件操作 import fs from "fs"; // 读取二进制文件 try { let buffer = fs.readFileSync("image.jpg"); print("二进制文件大小: " + buffer.length + " 字节"); } catch (e) { print("读取二进制文件时发生错误: " + e.message); } // 写入二进制文件 let data = new Buffer([0x48, 0x65, 0x6c, 0x6c, 0x6f]); // "Hello" 的 ASCII 编码 try { fs.writeFileSync("binary.bin", data); print("二进制文件写入成功"); } catch (e) { print("写入二进制文件时发生错误: " + e.message); }

文件系统路径

你可以使用 path 模块来处理文件路径。

// 文件路径操作 import path from "path"; // 连接路径 let fullPath = path.join("directory", "subdirectory", "file.txt"); print("完整路径: " + fullPath); // 获取文件名 let filename = path.basename("/path/to/file.txt"); print("文件名: " + filename); // 获取目录名 let dirname = path.dirname("/path/to/file.txt"); print("目录名: " + dirname); // 获取文件扩展名 let extname = path.extname("file.txt"); print("文件扩展名: " + extname); // 解析路径 let parsedPath = path.parse("/path/to/file.txt"); print("解析后的路径:"); print(parsedPath);

文件操作最佳实践

  • 总是使用 try-catch 块来处理文件操作可能的异常
  • 对于大型文件，优先使用异步操作以避免阻塞主线程
  • 使用 path 模块来处理跨平台的路径问题
  • 在操作完成后关闭文件句柄（对于流式操作）
  • 检查文件或目录是否存在后再进行操作
  • 使用适当的编码格式读取文本文件
// 文件操作最佳实践示例 import fs from "fs"; import path from "path"; def readFileSafely(filePath: string) -> string { // 确保路径存在 if (!fs.existsSync(filePath)) { throw new Error(`文件不存在: ${filePath}`); } // 检查是否为文件 let stats = fs.statSync(filePath); if (!stats.isFile()) { throw new Error(`路径不是文件: ${filePath}`); } // 读取文件 try { return fs.readFileSync(filePath, "utf8"); } catch (e) { throw new Error(`读取文件时发生错误: ${e.message}`); } } def writeFileSafely(filePath: string, content: string) { // 确保目录存在 let dir = path.dirname(filePath); if (!fs.existsSync(dir)) { fs.mkdirSync(dir, { recursive: true }); } // 写入文件 try { fs.writeFileSync(filePath, content); print(`文件写入成功: ${filePath}`); } catch (e) { throw new Error(`写入文件时发生错误: ${e.message}`); } } // 使用示例 try { let content = readFileSafely("input.txt"); print("读取的内容:"); print(content); writeFileSafely("output/new-file.txt", "处理后的内容: " + content); } catch (e) { print("错误: " + e.message); }

练习

尝试完成以下练习，以巩固你对文件 I/O 的理解：

  1. 创建一个函数，读取文件内容并计算单词数量
  2. 创建一个函数，将数组数据写入到 CSV 文件
  3. 创建一个函数，复制文件从源路径到目标路径
  4. 创建一个函数，列出目录及其子目录中的所有文件
// 练习答案示例 // 1. 计算文件中的单词数量 import fs from "fs"; def countWords(filename: string) -> int { try { let content = fs.readFileSync(filename, "utf8"); // 分割单词并过滤空字符串 let words = content.split(/\s+/).filter(word => word.length > 0); return words.length; } catch (e) { print("错误: " + e.message); return 0; } } print("单词数量: " + countWords("example.txt")); // 2. 写入 CSV 文件 import fs from "fs"; def writeCSV(filename: string, data: array>) { try { // 转换为 CSV 格式 let csvContent = data.map(row => row.join(",")).join("\n"); fs.writeFileSync(filename, csvContent); print("CSV 文件写入成功: " + filename); } catch (e) { print("错误: " + e.message); } } let data = [ ["Name", "Age", "City"], ["John", "30", "New York"], ["Jane", "25", "London"], ["Bob", "35", "Paris"] ]; writeCSV("people.csv", data); // 3. 复制文件 import fs from "fs"; def copyFile(source: string, destination: string) { try { // 读取源文件 let content = fs.readFileSync(source); // 写入目标文件 fs.writeFileSync(destination, content); print("文件复制成功: " + source + " -> " + destination); } catch (e) { print("错误: " + e.message); } } copyFile("source.txt", "destination.txt"); // 4. 列出目录中的所有文件 import fs from "fs"; import path from "path"; def listAllFiles(dir: string) -> array { let files: array = []; function traverse(currentDir: string) { let entries = fs.readdirSync(currentDir); for (let entry in entries) { let fullPath = path.join(currentDir, entry); let stats = fs.statSync(fullPath); if (stats.isFile()) { files.push(fullPath); } else if (stats.isDirectory()) { traverse(fullPath); } } } traverse(dir); return files; } let allFiles = listAllFiles("."); print("目录中的所有文件:"); for (let file in allFiles) { print(file); }
