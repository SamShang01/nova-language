模块

在 Nova 中创建和使用模块

模块的创建

在 Nova 中，你可以将代码组织到模块中，以便更好地管理和重用代码。

// math.nova - 数学工具模块 def add(a: int, b: int) -> int { return a + b; } def subtract(a: int, b: int) -> int { return a - b; } def multiply(a: int, b: int) -> int { return a * b; } def divide(a: int, b: int) -> float { if (b == 0) { throw new Error("除数不能为零"); } return a / b; } // 导出函数 export { add, subtract, multiply, divide };

模块的导入

你可以使用 import 关键字来导入模块中的函数和变量。

// 导入整个模块 import * as math from "./math"; print(math.add(5, 3)); // 输出: 8 print(math.subtract(10, 4)); // 输出: 6 // 导入特定函数 import { add, multiply } from "./math"; print(add(2, 3)); // 输出: 5 print(multiply(4, 5)); // 输出: 20 // 导入并重命名 import { divide as divideNumbers } from "./math"; print(divideNumbers(10, 2)); // 输出: 5.0

模块的导出

你可以使用 export 关键字来导出模块中的函数、变量和类。

// utils.nova - 工具模块 // 导出变量 export let PI = 3.14159; export let E = 2.71828; // 导出函数 export def greet(name: string) -> string { return "Hello, " + name + "!"; } // 导出类 export class Calculator { def add(a: int, b: int) -> int { return a + b; } def subtract(a: int, b: int) -> int { return a - b; } } // 导出默认值 export default def sayHello() { print("Hello from default export!"); }

默认导出

每个模块可以有一个默认导出，使用 export default 关键字。

// logger.nova - 日志模块 export default class Logger { def log(message: string) { print("[LOG] " + message); } def error(message: string) { print("[ERROR] " + message); } def info(message: string) { print("[INFO] " + message); } } // 导入默认导出 import Logger from "./logger"; let logger = new Logger(); logger.log("This is a log message"); logger.error("This is an error message"); logger.info("This is an info message");

目录结构

对于较大的项目，你可以使用目录来组织模块。

目录结构示例

project/ ├── src/ │ ├── main.nova │ ├── math/ │ │ ├── index.nova │ │ ├── arithmetic.nova │ │ └── geometry.nova │ └── utils/ │ ├── index.nova │ └── helpers.nova └── package.json

index.nova 文件

在目录中创建 index.nova 文件，可以将目录作为模块导入。

// math/index.nova - 数学模块入口 import { add, subtract, multiply, divide } from "./arithmetic"; import { calculateArea, calculatePerimeter } from "./geometry"; export { add, subtract, multiply, divide, calculateArea, calculatePerimeter }; // 导入整个数学模块 import * as math from "./math"; print(math.add(5, 3)); print(math.calculateArea(5)); // 计算圆的面积

内置模块

Nova 提供了一些内置模块，用于常见的功能。

// 使用内置模块 // 文件系统模块 import fs from "fs"; // 路径模块 import path from "path"; // 操作系统模块 import os from "os"; // 网络模块 import net from "net"; // 示例：使用 fs 模块 let content = fs.readFileSync("example.txt", "utf8"); print(content); // 示例：使用 path 模块 let fullPath = path.join("directory", "file.txt"); print(fullPath); // 示例：使用 os 模块 print("操作系统: " + os.platform()); print("CPU 核心数: " + os.cpus().length);

模块的循环依赖

当两个或多个模块相互导入时，会产生循环依赖。在 Nova 中，你需要小心处理这种情况。

// moduleA.nova import { functionB } from "./moduleB"; export def functionA() { print("Function A"); functionB(); } // moduleB.nova import { functionA } from "./moduleA"; export def functionB() { print("Function B"); // 注意：这里调用 functionA 会导致无限递归 // functionA(); } // main.nova import { functionA } from "./moduleA"; import { functionB } from "./moduleB"; functionA(); // 输出: Function A, Function B functionB(); // 输出: Function B

模块的缓存

当你导入一个模块时，Nova 会缓存该模块的导出值。这意味着模块只会被执行一次。

// counter.nova let count = 0; print("模块被加载"); export def increment() { count++; return count; } export def getCount() { return count; } // main1.nova import { increment, getCount } from "./counter"; print(increment()); // 输出: 1 print(getCount()); // 输出: 1 // main2.nova import { increment, getCount } from "./counter"; print(increment()); // 输出: 2 (模块被缓存，count 保持之前的值) print(getCount()); // 输出: 2

模块最佳实践

  • 将相关功能组织到同一个模块中
  • 使用目录结构来组织大型项目
  • 使用 index.nova 文件作为模块入口
  • 避免循环依赖
  • 使用明确的导出和导入语句
  • 为模块添加适当的文档
// 模块最佳实践示例 // 1. 组织相关功能 // utils/string.nova - 字符串工具 export def capitalize(str: string) -> string { if (str.length === 0) return str; return str.charAt(0).toUpperCase() + str.slice(1); } export def truncate(str: string, length: int) -> string { if (str.length <= length) return str; return str.slice(0, length) + "..."; } // 2. 使用 index.nova 作为入口 // utils/index.nova import * as string from "./string"; import * as number from "./number"; export { string, number }; // 3. 明确的导入 export { add, subtract } from "./arithmetic"; export { calculateArea } from "./geometry";

练习

尝试完成以下练习，以巩固你对模块的理解：

  1. 创建一个 string 模块，包含字符串处理函数
  2. 创建一个 array 模块，包含数组处理函数
  3. 创建一个目录结构，包含多个模块，并使用 index.nova 作为入口
  4. 创建一个使用内置模块的程序
// 练习答案示例 // 1. string 模块 // string.nova export def capitalize(str: string) -> string { if (str.length === 0) return str; return str.charAt(0).toUpperCase() + str.slice(1); } export def reverse(str: string) -> string { return str.split("").reverse().join(""); } export def countWords(str: string) -> int { return str.split(/\s+/).filter(word => word.length > 0).length; } export def toCamelCase(str: string) -> string { return str.split("-") .map((word, index) => index === 0 ? word : word.charAt(0).toUpperCase() + word.slice(1)) .join(""); } export { capitalize, reverse, countWords, toCamelCase }; // 2. array 模块 // array.nova export def find(array: array, predicate: function) -> any { for (let item in array) { if (predicate(item)) { return item; } } return null; } export def filter(array: array, predicate: function) -> array { let result = []; for (let item in array) { if (predicate(item)) { result.push(item); } } return result; } export def map(array: array, mapper: function) -> array { let result = []; for (let item in array) { result.push(mapper(item)); } return result; } export def reduce(array: array, reducer: function, initialValue: any) -> any { let accumulator = initialValue; for (let item in array) { accumulator = reducer(accumulator, item); } return accumulator; } export { find, filter, map, reduce }; // 3. 目录结构 /* utils/ ├── index.nova ├── string.nova └── array.nova */ // utils/index.nova import * as string from "./string"; import * as array from "./array"; export { string, array }; // 使用 import { string, array } from "./utils"; print(string.capitalize("hello")); print(array.map([1, 2, 3], x => x * 2)); // 4. 使用内置模块 import fs from "fs"; import path from "path"; // 读取目录内容 let files = fs.readdirSync("."); print("当前目录文件:"); for (let file in files) { print(file); } // 连接路径 let filePath = path.join(".", "example.txt"); print("文件路径: " + filePath); // 读取文件内容 if (fs.existsSync(filePath)) { let content = fs.readFileSync(filePath, "utf8"); print("文件内容:"); print(content); } else { print("文件不存在"); }
